Algorithm Pseudocode

procedure selectionSort(A : list)
  for i = 0 to length(A) - 1 do
    minIdx = i
    for j = i + 1 to length(A) - 1 do
      if A[j] < A[minIdx] then
        minIdx = j
    if minIdx != i then
      swap(A[i], A[minIdx])
