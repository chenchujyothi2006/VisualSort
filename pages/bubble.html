Algorithm Pseudocode

procedure bubbleSort(A : list)

  n = length(A)

  for i = 0 to n - 1 do

    for j = 0 to n - i - 2 do

      if A[j] > A[j + 1] then

        swap(A[j], A[j + 1])

  mark A[n - i - 1] as sorted
